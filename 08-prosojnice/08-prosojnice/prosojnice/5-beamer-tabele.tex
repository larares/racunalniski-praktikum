\begin{frame}{Odkrivanje tabele po vrsticah}
	Včasih pride prav, da tabelo odkrivamo postopoma po vrsticah.
	\begin{center}
		\begin{tabular}{c|cccc}
		   Oznaka & A & B & C & D \\ \hline
		   \pause
		   X & 1 & 2 & 3 & 4 \\
		   \pause
		   Y & 3 & 4 & 5 & 6 \\
		   \pause
		   Z & 5 & 6 & 7 & 8
		\end{tabular}
	\end{center}
\end{frame}
 

\begin{frame}{Odkrivanje tabele po stolpcih}

	\begin{center}
		\begin{tabular}{c|>{\onslide<2->}cccc<{\onslide}}
		   \onslide<1-5>{Oznaka} & \onslide<2-5>{A} & \onslide<3-5>{B} & \onslide<4-5>{C} & \onslide<5>{D} \\ \hline
		   \onslide<1-5>{X} & \onslide<2-5>{1} & \onslide<3-5>{2} & \onslide<4-5>{3} & \onslide<5>{4} \\

		\end{tabular}
	\end{center}
\end{frame}
